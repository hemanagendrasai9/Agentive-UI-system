    <script>

        // Add basic interactivity

        document.addEventListener('DOMContentLoaded', function() {

            // Make nodes draggable

            const nodes = document.querySelectorAll('.workflow-node');

            let draggedNode = null;

            let offset = { x: 0, y: 0 };

           

            nodes.forEach(node => {

                node.addEventListener('mousedown', function(e) {

                    draggedNode = this;

                    const rect = this.getBoundingClientRect();

                    offset.x = e.clientX - rect.left;

                    offset.y = e.clientY - rect.top;

                    this.style.zIndex = '1000';

                   

                    // Remove selection from other nodes

                    nodes.forEach(n => n.classList.remove('selected'));

                    this.classList.add('selected');

                });

            });

           

            document.addEventListener('mousemove', function(e) {

                if (draggedNode) {

                    const canvas = document.querySelector('.workflow-canvas');

                    const canvasRect = canvas.getBoundingClientRect();

                   

                    draggedNode.style.left = (e.clientX - canvasRect.left - offset.x) + 'px';

                    draggedNode.style.top = (e.clientY - canvasRect.top - offset.y) + 'px';

                }

            });

           

            document.addEventListener('mouseup', function() {

                if (draggedNode) {

                    draggedNode.style.zIndex = '';

                    draggedNode = null;

                }

            });

           

            // Test workflow button

            const testBtn = document.querySelector('.btn-primary');

            testBtn.addEventListener('click', function() {

                // Simulate workflow execution

                const statusElements = document.querySelectorAll('.node-status');

                statusElements.forEach((status, index) => {

                    setTimeout(() => {

                        status.className = 'node-status status-running';

                        setTimeout(() => {

                            status.className = 'node-status status-success';

                        }, 1000);

                    }, index * 500);

                });

            });

           

            // Palette item drag simulation

            const paletteItems = document.querySelectorAll('.node-item');

            paletteItems.forEach(item => {

                item.addEventListener('dragstart', function(e) {

                    e.dataTransfer.setData('text/plain', this.querySelector('.node-name').textContent);

                });

            });

        });

    </script>
